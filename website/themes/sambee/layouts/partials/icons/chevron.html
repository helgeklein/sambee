{{/* Purpose: Renders the shared chevron icon with configurable direction or rotation. */}} {{- $baseClass := "size-[1em] inline-block align-[-0.125em]" -}} {{- $class := .class | default $baseClass -}} {{- $direction := (.direction | default "right") | lower -}} {{- $rotationByDirection := dict "right" 0 "down" 90 "left" 180 "up" -90 -}} {{- $rotation := index $rotationByDirection $direction | default 0 -}} {{- with .rotate -}} {{- $rotation = . -}} {{- end -}} {{- $transform := "" -}} {{- if ne $rotation 0 -}} {{- $transform = printf "rotate(%v 10 10)" $rotation -}} {{- end -}}
